function limit(fun, max_v) {
	
  function aux_fun(...values) {
  	if(aux_fun.max == 0){
        	return undefined
    }
    else {
			aux_fun.max--;
    	return aux_fun.fun(...values)
    }
  }
  aux_fun.max = max_v
  aux_fun.fun = fun
	return aux_fun
}

let limitAdd = limit(add, 2);
console.log(limitAdd(3,5));// 8
console.log(limitAdd(11, 23)); //34
console.log(limitAdd(5, 10)); // undefined

 return x + y;
